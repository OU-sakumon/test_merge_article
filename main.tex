% 目次の「章（chapter）」のデザイン定義
\titlecontents{chapter}[0pt]                 % 左端の余白
  {\addvspace{1.5em}}                        % 各項目の上の間隔
  {                                          % 【番号付きの章のデザイン】
    \begin{tikzpicture}[remember picture, baseline={(0, -0.1)}]
      % 章番号の入るボックス（影付き）
      \node[
        fill=black!10,                       % 背景色（薄いグレー）
        draw=none,                           % 枠線なし
        rounded corners=3pt,                 % 角を丸く
        inner sep=6pt,                       % 内側の余白
        font=\sffamily\bfseries\large,       % フォント設定
        drop shadow={opacity=0.2, shadow xshift=1pt, shadow yshift=-1pt} % 影
      ] (num) {第\thecontentslabel 章};
    \end{tikzpicture}
    \hspace{1em}                             % 番号とタイトルの間のスペース
    \bfseries\large\sffamily                 % タイトルのフォント設定
  }
  {\bfseries\large\sffamily}                                         % 
  {\hfill \bfseries \thecontentspage}        % ページ番号（右端に太字で）

% #1: タイトル
% #2: 所属
% #3: 名前
% #4: 前提知識
% #5: 概要
\newcommand{\CustomChapter}[5]{
  % 1. 章番号
\refstepcounter{chapter}
  \markboth{第\thechapter 章\quad #1}{第\thechapter 章\quad #1}
  \addcontentsline{toc}{chapter}{\protect\numberline{\thechapter}#1
  \hspace{0em}
    {：#3}% 名前
    \vspace{2mm}
    \newline
    \hspace{22mm}
    {\small 前提知識：#4}
  }
  \noindent
  \begin{center}{{\huge\sffamily 第\thechapter 章} }\end{center}
  \par            
  \vspace{0.5em}   % 章番号とタイトルの間の少しの隙間
  
  % 2. タイトル（下の行に表示）
  \noindent
  \begin{center}{{\Huge\sffamily #1} }\end{center}
  \par
  
  \vspace{1cm}
  
  % 2. 所属と名前
  \begin{center}
    {\Large
    \begin{tabular}{ll}
      所\quad 属 \quad ：& #2 \\[1em]
      名\quad 前 \quad ：& #3 \\
    \end{tabular}
    }
  \end{center}
  
  \vspace{1cm}
  \noindent
  {\large\sffamily\textbf{■ 前提知識}} \\
  \vspace{0.2em}
  #4
  \par
  \vspace{1mm}
  
  \noindent
  {\large\sffamily\textbf{■ 概 要}} \\
  \vspace{0.2em}
  #5
}
